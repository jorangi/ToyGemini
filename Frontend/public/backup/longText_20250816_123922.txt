                <h2>카라스노</h2>
                <div id="karasuno-score" class="score">0</div>
                <div class="buttons">
                    <button onclick="changeScore('karasuno', 1)">+1</button>
                    <button onclick="changeScore('karasuno', -1)">-1</button>
                </div>
            </div>
            <div class="team inarizaki">
                <h2>이나리자키</h2>
                <div id="inarizaki-score" class="score">0</div>
                <div class="buttons">
                    <button onclick="changeScore('inarizaki', 1)">+1</button>
                    <button onclick="changeScore('inarizaki', -1)">-1</button>
                </div>
            </div>
        </div>
        <button class="reset-button" onclick="resetScores()">점수 초기화</button>
    </div>

    <script>
        let karasunoScore = 0;
        let inarizakiScore = 0;

        function changeScore(team, value) {
            if (team === 'karasuno') {
                karasunoScore += value;
                if (karasunoScore < 0) karasunoScore = 0;
                document.getElementById('karasuno-score').innerText = karasunoScore;
            } else if (team === 'inarizaki') {
                inarizakiScore += value;
                if (inarizakiScore < 0) inarizakiScore = 0;
                document.getElementById('inarizaki-score').innerText = inarizakiScore;
            }
        }

        function resetScores() {
            karasunoScore = 0;
            inarizakiScore = 0;
            document.getElementById('karasuno-score').innerText = 0;
            document.getElementById('inarizaki-score').innerText = 0;
        }
    </script>
